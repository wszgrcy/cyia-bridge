import { Component, viewChildren } from '@angular/core';
import {
  ComponentWrapper,
  ReactOutlet,
  portalChildren,
} from '@cyia/ngx-bridge';
import { useCurrentEditor } from '@tiptap/react';
import { Fragment, useMemo } from 'react';
@Component({
  selector: 'app-menubar',
  standalone: true,
  imports: [ReactOutlet],
  templateUrl: './menubar.component.html',
  styleUrl: './menubar.component.scss',
})
export class MenubarComponent implements ComponentWrapper {
  Fragment = Fragment;
  outletList = viewChildren<ReactOutlet>('portal');
  children = portalChildren(this.outletList);

  fragmentCustom = () => {
    const { editor } = useCurrentEditor();
    return useMemo(() => {
      if (!editor) {
        return { props: {} };
      }
      return {
        props: {},
        output: { editor },
      };
    }, [editor]);
  };
  ngOnInit(): void {}
}
