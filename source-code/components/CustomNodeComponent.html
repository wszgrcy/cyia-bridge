import {
  Component,
  SimpleChanges,
  computed,
  effect,
  input,
  signal,
  viewChildren,
} from '@angular/core';
import {
  ReactComponentWrapper,
  ReactOutlet,
  portalChildren,
} from '@cyia/ngx-bridge';
import { Handle, Position } from 'reactflow';

@Component({
  selector: 'app-custom-node',
  standalone: true,
  imports: [ReactOutlet],
  templateUrl: './custom-node.component.html',
  styleUrl: './custom-node.component.scss',
})
export class CustomNodeComponent implements ReactComponentWrapper {
  Handle = Handle;
  Position = Position;
  props = input<any>();
  outletList = viewChildren<ReactOutlet>('portal');
  children = portalChildren(this.outletList);
  output = signal<any>(undefined);
  constructor() {
    effect(() => {
      let list = this.outletList();
      let output = this.output();
      for (const item of list) {
        let portal = item.inited();
        if (!portal) {
          return undefined;
        }
      }
      if (output) {
        output['update']();
      }
    });
  }
  inputChanged(value: string) {
    console.log(value);
  }
  ngOnChanges(changes: SimpleChanges): void {
    // console.log('input', this.props());
  }
  ngOnDestroy(): void {
    //Called once, before the instance is destroyed.
    //Add 'implements OnDestroy' to the class.
    console.log('CustomNodeComponent-ngOnDestroy');
  }
}
