import { Component } from '@angular/core';
import { ReactOutlet } from '@cyia/ngx-bridge';
import { Fragment } from 'react';
import { useForm, SubmitHandler } from 'react-hook-form';
import { InputDirective } from './input.directive';
import { PurePipe } from './pure.pipe';
import { JsonPipe } from '@angular/common';
type Inputs = {
  example: string;
  exampleRequired: string;
};
@Component({
  selector: 'app-react-hook-form-demo',
  standalone: true,
  imports: [ReactOutlet, InputDirective, PurePipe, JsonPipe],
  templateUrl: './react-hook-form-demo.component.html',
  styleUrl: './react-hook-form-demo.component.scss',
})
export class ReactHookFormDemoComponent {
  scope = Fragment;
  context = () => {
    const {
      register,
      handleSubmit,
      watch,
      formState: { errors },
    } = useForm<Inputs>({
      defaultValues: {
        example: '',
        exampleRequired: '',
      },
    });

    console.log(watch('example')); // watch input value by passing the name of it
    return {
      props: {},
      output: {
        handleSubmit: handleSubmit((data) => {
          alert(JSON.stringify(data));
        }),
        register,
        errors,
      },
    };
  };
}
