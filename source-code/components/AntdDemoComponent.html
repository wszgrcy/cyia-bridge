import { Component } from '@angular/core';
import { ReactOutlet, wrapperToReact } from '@cyia/ngx-bridge';
import { Flex, Button, Tree, TreeDataNode, TreeProps } from 'antd';
import { CustomTitleComponent } from './custom-title.component';
import { useMemo } from 'react';
const treeData: TreeDataNode[] = [
  {
    title: 'parent 1',
    key: '0-0',
    children: [
      {
        title: 'parent 1-0',
        key: '0-0-0',
        disabled: true,
        children: [
          {
            title: 'leaf',
            key: '0-0-0-0',
            disableCheckbox: true,
          },
          {
            title: 'leaf',
            key: '0-0-0-1',
          },
        ],
      },
      {
        title: 'parent 1-1',
        key: '0-0-1',
        children: [
          {
            title: wrapperToReact(CustomTitleComponent).reactElement,
            key: '0-0-1-0',
          },
        ],
      },
    ],
  },
];
@Component({
  selector: 'app-antd-demo',
  standalone: true,
  imports: [ReactOutlet],
  templateUrl: './antd-demo.component.html',
  styleUrl: './antd-demo.component.scss',
})
export class AntdDemoComponent {
  Flex = Flex;
  Button = Button;
  Tree = Tree;
  testClick = (event: any) => {
    console.log(event);
    console.log(this);
  };
  treeContext = () => {
    const onSelect: TreeProps['onSelect'] = (selectedKeys, info) => {
      console.log('selected', selectedKeys, info);
    };

    const onCheck: TreeProps['onCheck'] = (checkedKeys, info) => {
      console.log('onCheck', checkedKeys, info);
    };
    return useMemo(
      () => ({
        props: {
          checkable: true,
          defaultExpandedKeys: ['0-0-0', '0-0-1'],
          defaultSelectedKeys: ['0-0-0', '0-0-1'],
          defaultCheckedKeys: ['0-0-0', '0-0-1'],
          onSelect: onSelect,
          onCheck: onCheck,
          treeData: treeData,
        },
      }),
      []
    );
  };
}
