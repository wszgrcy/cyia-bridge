import { Component } from '@angular/core';
import { Transition } from 'vue';
import { VueOutlet } from '@cyia/ngx-bridge';
import gsap from 'gsap';
import { BridgeChild } from '@cyia/ngx-bridge/common';

function onBeforeEnter(el: any) {
  gsap.set(el, {
    scaleX: 0.25,
    scaleY: 0.25,
    opacity: 1,
  });
}

function onEnter(el: any, done: any) {
  gsap.to(el, {
    duration: 1,
    scaleX: 1,
    scaleY: 1,
    opacity: 1,
    ease: 'elastic.inOut(2.5, 1)',
    onComplete: done,
  });
}

function onLeave(el: any, done: any) {
  gsap.to(el, {
    duration: 0.7,
    scaleX: 1,
    scaleY: 1,
    x: 300,
    ease: 'elastic.inOut(2.5, 1)',
  });
  gsap.to(el, {
    duration: 0.2,
    delay: 0.5,
    opacity: 0,
    onComplete: done,
  });
}
@Component({
  selector: 'app-vue-transition-demo',
  standalone: true,
  imports: [VueOutlet, BridgeChild],
  templateUrl: './vue-transition-demo.component.html',
  styleUrl: './vue-transition-demo.component.scss',
})
export class VueTransitionDemoComponent {
  Transition = Transition;
  show = true;
  show2 = true;
  onBeforeEnter = onBeforeEnter;
  onEnter = onEnter;
  onLeave = onLeave;
}

<div>
  <button (click)="show = !show">Toggle</button>

  <vue-outlet [component]="Transition" [root]="true">
    <bridge-child #child [enable]="show">
      <div #child class="real">
        <p>hello</p>
      </div>
    </bridge-child>
  </vue-outlet>
</div>
<div>
  <button (click)="show2 = !show2">Toggle2</button>
  <vue-outlet
    [component]="Transition"
    [root]="true"
    [props]="{
      name: 'box',

    }"
    style="display: flex; margin-top: 20px"
  >
    <bridge-child #child [enable]="show2">
      <div #child class="gsap-box"></div>
    </bridge-child>
  </vue-outlet>
</div>
