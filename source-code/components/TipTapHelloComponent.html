import { Component, ViewChild, viewChild } from '@angular/core';
import { ReactOutlet } from '@cyia/ngx-bridge';
import { EditorProvider, FloatingMenu, BubbleMenu } from '@tiptap/react';
import StarterKit from '@tiptap/starter-kit';
import { createElement } from 'react';
const extensions = [StarterKit];
const content = '<p>Hello World!</p>';

@Component({
  selector: 'app-hello',
  standalone: true,
  imports: [ReactOutlet],
  templateUrl: './hello.component.html',
  styleUrl: './hello.component.scss',
})
export class TipTapHelloComponent {
  EditorProvider = (props: any) => {
    return createElement(
      'div',
      null,
      createElement(EditorProvider, props, props.children)
    );
  };
  FloatingMenu = FloatingMenu;
  BubbleMenu = BubbleMenu;
  editorProps = { extensions, content };
}

<react-outlet
  [component]="EditorProvider"
  [root]="true"
  [props]="$any(editorProps)"
>
  <react-outlet [component]="FloatingMenu" #child>
    <ng-template #child>This is the floating menu</ng-template>
  </react-outlet>
  <react-outlet [component]="BubbleMenu" #child>
    <ng-template #child>This is the bubble menu</ng-template>
