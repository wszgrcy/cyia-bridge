import { Component, viewChild } from '@angular/core';
import { Editable, withReact, Slate } from 'slate-react';
import { Toolbar } from './components';
import {
  BlockButton,
  MarkButton,
  Element,
  Leaf,
  initialValue,
  toggleMark,
  HOTKEYS,
} from './a';
import { ReactOutlet } from '@cyia/ngx-bridge';
import { createElement, useCallback, useContext, useMemo } from 'react';
import { withHistory } from 'slate-history';
import { createEditor } from 'slate';
import isHotkey from 'is-hotkey';

@Component({
  selector: 'app-rich-text',
  standalone: true,
  imports: [ReactOutlet],
  templateUrl: './rich-text.component.html',
  styleUrl: './rich-text.component.scss',
})
export class RichTextComponent {
  root = viewChild.required<ReactOutlet>('root');
  Slate = Slate;
  Toolbar = Toolbar;
  MarkButton = MarkButton;
  BlockButton = BlockButton;
  Editable = Editable;
  slateContext = () => {
    const renderElement = useCallback(
      (props: any) => createElement(Element, props),
      []
    );
    const renderLeaf = useCallback(
      (props: any) => createElement(Leaf, props),
      []
    );
    const editor = useMemo(() => withHistory(withReact(createEditor())), []);
    return {
      props: {
        editor,
        initialValue,
      },
      output: {
        renderElement,
        renderLeaf,
        editor,
      },
    };
  };
  editableContext = () => {
    let root = useContext(this.root().CONTEXT);
    return {
      props: {
        renderElement: root['renderElement'],
        renderLeaf: root['renderLeaf'],
        placeholder: 'Enter some rich textâ€¦',
        spellCheck: true,
        autoFocus: true,
        editor: root['editor'],
        onKeyDown: (event: any) => {
          for (const hotkey in HOTKEYS) {
            if (isHotkey(hotkey, event as any)) {
              event.preventDefault();
              const mark = (HOTKEYS as any)[hotkey];
              toggleMark(root['editor'], mark);
            }
          }
        },
      },
    };
  };
}
