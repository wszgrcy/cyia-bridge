import { Component, EnvironmentInjector, inject } from '@angular/core';
import { ReactOutlet, wrapperToReact } from '@cyia/ngx-bridge';
import { createElement, useMemo } from 'react';
import {
  ReactFlow,
  MiniMap,
  useNodesState,
  useEdgesState,
  Background,
  Controls,
  useUpdateNodeInternals,
} from 'reactflow';
import { NgAComponent } from '../ng-a.component';
import { CustomNodeComponent } from '../custom-node/custom-node.component';
import { defaultNodes, initialEdges } from './nodes-edges';
import { RCustom2 } from './custom-node2';

@Component({
  selector: 'app-custom-node-examples',
  standalone: true,
  imports: [ReactOutlet],
  templateUrl: './custom-node-examples.component.html',
  styleUrl: './custom-node-examples.component.scss',
})
export class CustomNodeExamplesComponent {
  #environmentInjector = inject(EnvironmentInjector);

  // 直接调用
  // ReactFlow = ReactFlow
  // 需要上下文调用,可以改 component 也可以改 props 都带上下文
  ReactFlow = (props: any) => {
    const [nodes, setNodes, onNodesChange] = useNodesState(props.nodes);
    const [edges, setEdges, onEdgesChange] = useEdgesState(props.edges);

    return createElement(
      ReactFlow,
      { ...props, nodes, edges, onNodesChange, onEdgesChange },
      props.children
    );
  };
  MiniMap = MiniMap;
  Background = Background;
  Controls = Controls;

  switchValue = true;

  custom = () => {
    return {
      props: {
        nodes: defaultNodes,
        edges: initialEdges,
        nodeTypes: useMemo(
          () => ({
            customNode: wrapperToReact(
              CustomNodeComponent,
              {
                environmentInjector: this.#environmentInjector,
              },
              (props) => {
                const updateNodeInternals = useUpdateNodeInternals();
                return useMemo(
                  () => ({
                    output: {
                      update: () => {
                        updateNodeInternals(props.id);
                      },
                    },
                  }),
                  [props]
                );
              }
            ).reactFunctionComponent,
            customNode2: RCustom2,
          }),
          []
        ),
      },
    };
  };
  reactBProps = () => {
    return {
      el: wrapperToReact(NgAComponent, {
        environmentInjector: this.#environmentInjector,
      }).reactElement,
    };
  };
  switch() {
    this.switchValue = !this.switchValue;
  }
  miniMapConfig = { zoomable: true, pannable: true };
  changMiniMapConfig() {
    this.miniMapConfig = { zoomable: false, pannable: false };
  }
}

<react-outlet
  [component]="ReactFlow"
  style="display: block; height: 500px"
  [root]="true"
  [runInReact]="custom"
>
  @if (switchValue) {

  <react-outlet
    [component]="MiniMap"
    #child
    [props]="miniMapConfig"
  ></react-outlet>
  }
  <react-outlet [component]="Background" #child></react-outlet>
  <react-outlet [component]="Controls" #child></react-outlet>
</react-outlet>
