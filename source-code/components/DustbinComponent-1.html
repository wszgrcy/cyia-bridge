import {
  Component,
  ElementRef,
  effect,
  signal,
  viewChild,
} from '@angular/core';
import { ComponentInReact, ComponentWrapper } from '@cyia/ngx-bridge';
import { ItemTypes } from '../types';
import { useDrop } from 'react-dnd';
import { CSSProperties } from 'react';

const style: CSSProperties = {
  height: '12rem',
  width: '12rem',
  marginRight: '1.5rem',
  marginBottom: '1.5rem',
  color: 'white',
  padding: '1rem',
  textAlign: 'center',
  fontSize: '1rem',
  lineHeight: 'normal',
  float: 'left',
};
@Component({
  selector: 'app-dustbin',
  standalone: true,
  imports: [],
  templateUrl: './dustbin.component.html',
  styleUrl: './dustbin.component.scss',
})
export class DustbinComponent extends ComponentInReact {
  runInReact = () => {
    const [{ canDrop, isOver }, drop] = useDrop(() => ({
      accept: ItemTypes.BOX,
      drop: () => ({ name: 'Dustbin' }),
      collect: (monitor) => ({
        isOver: monitor.isOver(),
        canDrop: monitor.canDrop(),
      }),
    }));

    const isActive = canDrop && isOver;
    let backgroundColor = '#222';
    if (isActive) {
      backgroundColor = 'darkgreen';
    } else if (canDrop) {
      backgroundColor = 'darkkhaki';
    }
    return {
      output: {
        drop,
        style: { ...style, backgroundColor },
        isActive,
      },
    };
  };
  item = viewChild.required<ElementRef<HTMLElement>>('item');
  override output = signal<Record<string, any> | undefined>(undefined);
  constructor() {
    super();
    effect(() => {
      let elRef = this.item();
      let output = this.output();
      if (output) {
        output['drop'](elRef.nativeElement);
      }
    });
  }
}
